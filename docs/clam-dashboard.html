Clam Dashboard

Auto-updated from plate exports in data/clam

What This Shows

  • Plate-wide trajectories across time for every well.
  • Per-well trajectories for all wells (filterable by experiment in the static report by scrolling to the relevant facets).
  • QC table with parse status and warning flags.

Interactive Explorer

Use these selectors to focus on one experiment, one plate, or specific wells.

dashboard = await FileAttachment("output/clam/dashboard-data.csv").csv({typed: true})

experimentValues = [...new Set(dashboard.map(d => d.experiment_dir))]

experimentSortKey = (name) => {
  const m = /^([0-9]{8})/.exec(name)
  if (!m) return Number.NEGATIVE_INFINITY
  return Number.parseInt(m[1], 10)
}

mostRecentExperiment = experimentValues
  .slice()
  .sort((a, b) => {
    const ka = experimentSortKey(a)
    const kb = experimentSortKey(b)
    if (ka !== kb) return kb - ka
    return a.localeCompare(b)
  })[0]

experiments = ["All", ...experimentValues]
viewof experimentSel = Inputs.select(experiments, {
  label: "Experiment",
  value: mostRecentExperiment || "All"
})

plateCandidates = dashboard
  .filter(d => experimentSel === "All" || d.experiment_dir === experimentSel)
plates = ["All", ...new Set(plateCandidates.map(d => d.plate_id))]
viewof plateSel = Inputs.select(plates, {
  label: "Plate",
  value: "All"
})

wellCandidates = dashboard
  .filter(d => (experimentSel === "All" || d.experiment_dir === experimentSel) &&
               (plateSel === "All" || d.plate_id === plateSel))
wells = [...new Set(wellCandidates.map(d => d.well_id))].sort()
defaultWells = wells.slice(0, Math.min(6, wells.length))

viewof wellSel = Inputs.select(wells, {
  label: "Well(s)",
  multiple: true,
  value: defaultWells
})

filtered = dashboard.filter(d =>
  (experimentSel === "All" || d.experiment_dir === experimentSel) &&
  (plateSel === "All" || d.plate_id === plateSel) &&
  (wellSel.length === 0 || wellSel.includes(d.well_id))
)

Plot.plot({
  height: 420,
  grid: true,
  x: {label: "Time (hours)"},
  y: {label: "Fluorescence"},
  color: {legend: true, label: "Well"},
  marks: [
    Plot.line(filtered, {
      x: "time_hr",
      y: "value",
      stroke: "well_id",
      z: d => `${d.experiment_dir}|${d.plate_id}|${d.well_id}`
    }),
    Plot.dot(filtered, {
      x: "time_hr",
      y: "value",
      fill: "well_id",
      r: 2.5,
      tip: true
    })
  ]
})
